4.矩阵乘法

import java.util.Scanner;
//测试通过
public class Main {

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        //几行
        int row = sc.nextInt();
        //相当于m
        int m = sc.nextInt();
        int[][] first = new int[row][row];
        int[][] next = new int[row][row];
        int[][] result = new int[row][row];
        for (int i = 0; i < row; i++)
            for (int j = 0; j < row; j++){
                next[i][j]=first[i][j] = sc.nextInt();
            }

        //判断是否为0
        if(m==0){
            for(int i=0;i<row;i++)
                for(int j=0;j<row;j++)
                    if(i==j)
                        result[i][j]=1;
                    else
                        result[i][j]=0;
        }
        else
        for (int h = 0; h < m - 1; h++) {
            result = new int[row][row];
            for (int i = 0; i < row; i++) {
                for (int j = 0; j < row; j++) {
                    for (int k = 0; k < row; k++) {
                        result[i][j] += next[i][k] * first[k][j];
                    }
                }
            }
            next = result;
        }
        for (int i = 0; i < row; i++) {
            for (int j = 0; j < row; j++) {
                System.out.print(next[i][j] + "\t");
            }
            System.out.println();
        }
    }

}

8.Sine之舞

这个好像不用递归不好做吧。

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int n = scanner.nextInt();
        String string = f(n, 1, 1);
        System.out.println(string);
    }

    private static String f1(int n, int i, int j) {
        if (i == n) return "sin(" + j + ")";
        if (i % 2 == 0) return "sin(" + i + "+" + f1(n,
                i + 1, j + 1) + ")";
        else return "sin(" + i + "-" + f1(n, i + 1, j + 1) + ")";
    }

    private static String f(int n, int i, int j) {
        if (i == n) return f1(n, i, j) + "+" + i;
        return "(" + f(n, i + 1, j) + ")" + f1(n, i, j) + "+" + i;
    }
}

9.FJ的字符串

import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int t = sc.nextInt();
        print(t);
    }

    private static void print(int t) {
        if (t == 1)
            System.out.print("A");
        else if (t <= 26) {
            print(t - 1);
            System.out.print((char) (64 + t));
            print(t - 1);
        }
    }
}
